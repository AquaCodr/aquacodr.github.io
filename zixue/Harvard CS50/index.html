void judge(long int number);

int main()
{
    // 写入卡号n
    long int n = get_long("Number: ");

    // 判断卡号是否合法,合法时发卡机构
    judge(n);
}


void judge(long int number)
{
    // 计算Luhn数
    long int keep = number;
    long int card = number;
    int count = 0;
    while(number / 10 != 0)
    {
        int k = number / 10 % 10;
        if((0 <= k * 2) && (k*2<=9))
        {
            count = count + 2 * k;
        }
        else
        {
            count = count + (2 * k) % 10 + (2 * k) / 10;
        }
        number = number / 100;
    }
    while(keep != 0)
    {
        count = count + keep % 10;
        keep = keep / 100;
    }
    // 提取卡首位
    int m;
    while(card != 0)
    {
        m = card;
        card = card / 10;
    }

    // 判断合法性及银行
    if(count % 10 != 0)
    {
        printf("INVALID\n");
    }
    else if(m == 4)
    {
        printf("VISA\n");
    }
    else if(m == 3)
    {
        printf("AMEX\n");
    }
    else if(m == 5 )
    {
        printf("MASTERCARD\n");
    }
}

{
    float n = (bill * (1.00 + tax/100.00) * (1.00 + (float)tip / 100.00)) / 2.00;
